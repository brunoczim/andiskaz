Computes the overlapped area between two rectangles. An empty rectange is produced if both do not overlap.

Examples

use gardiz::{rect::Rect, coord::Vec2};

let left: Rect<u16> = Rect {
    start: Vec2 { x: 5, y: 3 },
    size: Vec2 { x: 10, y: 7 },
};
let right = Rect {
    start: Vec2 { x: 9, y: 4 },
    size: Vec2 { x: 9, y: 5 },
};
let overlapped = Rect {
    start: Vec2 { x: 9, y: 4 },
    size: Vec2 { x: 6, y: 5 },
};
assert_eq!(left.overlapped(&right), overlapped);
