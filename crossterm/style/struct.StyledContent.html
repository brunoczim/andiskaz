pub struct StyledContent<D: Display> { /* fields omitted */ }
Expand description

The style with the content to be styled.

Examples

use crossterm::style::{style, Color, Attribute};

let styled = style("Hello there")
    .with(Color::Yellow)
    .on(Color::Blue)
    .attribute(Attribute::Bold);

println!("{}", styled);
